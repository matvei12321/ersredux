is_chn_hehemony_hive = {
	custom_trigger_tooltip = {
		tooltip = is_chn_hehemony_hive_tt
		OR = {
			has_idea = changelings_no_puppet
			has_idea = changelings_low_puppet
			has_idea = changelings_medium_puppet
			has_idea = changelings_high_puppet
			has_idea = changelings_leader
		}
	}
}
